Google Sheets CRM Setup
Create your live CRM in Google Sheets
Paste this script into Google Apps Script to auto-create your full Ortho Dog pipeline CRM with all columns, color coding, and sample data.
function createOrthoDogCRM() { var ss = SpreadsheetApp.create('Ortho Dog — Live CRM'); // Pipeline sheet var ps = ss.getActiveSheet(); ps.setName('Pipeline'); ps.getRange('A1:J1').setValues([[ 'Lead Name','Company','Type','Stage', 'Target Score','Last Contact','Next Action', 'Notes','Reagan Flagged','Status' ]]); ps.getRange('A1:J1').setFontWeight('bold') .setBackground('#3D2B1F').setFontColor('#FFFFFF'); ps.setColumnWidth(1,180); ps.setColumnWidth(2,160); ps.setColumnWidth(7,200); ps.setColumnWidth(8,200); // Reviews sheet var rs = ss.insertSheet('Reviews'); rs.getRange('A1:F1').setValues([[ 'Customer','Product','Stars','Review Text', 'Date','Morgan Used' ]]); rs.getRange('A1:F1').setFontWeight('bold') .setBackground('#27500A').setFontColor('#FFFFFF'); // Orders sheet var os = ss.insertSheet('Orders'); os.getRange('A1:G1').setValues([[ 'Order ID','Customer','Product','Status', 'Ship Date','Fit Check Sent','Review Collected' ]]); os.getRange('A1:G1').setFontWeight('bold') .setBackground('#0C447C').setFontColor('#FFFFFF'); var url = ss.getUrl(); Logger.log('CRM created: ' + url); SpreadsheetApp.getUi().alert('CRM created! URL: ' + url); }
